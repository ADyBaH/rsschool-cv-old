Task: Sum all the numbers of a array, except the highest and the lowest element. 

function sumArray(array) {
  let answer = 0;
  if (array == null) {
    return answer;
  } else if(array.length < 3) {
    return answer;
  } else {
    array.sort(function(a, b){return a - b});
    for (let i = 1; i < array.length - 1; i++){
      answer += array[i];
    }
    return answer;
  }
}
